import {
  Controller,
  Get,
  Post,
  Body,
  Patch,
  Param,
  Delete,
  UseGuards,
  HttpCode,
  Query,
  UploadedFile,
} from '@nestjs/common';
import { PostService } from './post.service';
import { CreatePostDto } from './dto/create-post.dto';
import { UpdatePostDto } from './dto/update-post.dto';
import { ApiBearerAuth, ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { AuthGuard } from '@nestjs/passport';
import { CurrentUser } from '../auth/decorators/current-user.decorator';
import { User } from '@prisma/client';
import { Post as PostClass } from '../_gen/prisma-class/post';
import { PaginationDto } from '../profile/dto/pagination.dto';
import { ApiImageFile } from '../profile/decorators/api-file-decorator';
import { Image } from '../_gen/prisma-class/image';
import { SelectPostsDto } from './dto/select-post.dto';
import { SelectAllPostsDto } from './dto/select-all-posts.dto';
import { Bookmarks } from '../_gen/prisma-class/bookmarks';
import { BookmarkDto } from './dto/bookmark.dto';
import { PostFilterDto } from './dto/post-filter.dto';
import { GetPostDto } from './dto/get-post.dto';
import { GetPostsResponse } from './entities/getPostsResponse';
import { GetBookmarksResponse } from './entities/getBookmarksResponse';
import { SearchQueryDto } from './dto/create-search.dto';

@ApiTags('Posts')
@Controller('post')
export class PostController {
  constructor(private readonly postService: PostService) {}

  @ApiBearerAuth('JWT-auth')
  @UseGuards(AuthGuard('jwt'))
  @HttpCode(200)
  @ApiOkResponse({
    type: PostClass,
  })
  @Post()
  create(@Body() createPostDto: CreatePostDto, @CurrentUser() user: User) {
    return this.postService.create(createPostDto, user);
  }
  @ApiBearerAuth('JWT-auth')
  @UseGuards(AuthGuard('jwt'))
  @HttpCode(200)
  @ApiOkResponse({
    type: GetPostsResponse,
  })
  /**
   * Return All current user posts
   */
  @Get()
  findAllUserPosts(
    @CurrentUser() user: User,
    @Query() selectPostsDto: SelectPostsDto,
  ) {
    return this.postService.findAllUserPosts(
      user,
      selectPostsDto.userId,
      selectPostsDto.type,
      selectPostsDto.priority,
      selectPostsDto.take && +selectPostsDto.take,
      selectPostsDto.skip && +selectPostsDto.skip,
    );
  }
  /**
   * @return an uploaded image of a post
   */
  @ApiBearerAuth('JWT-auth')
  @UseGuards(AuthGuard('jwt'))
  @ApiOkResponse({ type: Image })
  @HttpCode(200)
  @ApiImageFile('image')
  @Post('/addImage')
  addImage(
    @UploadedFile()
    file: Express.Multer.File,
    @CurrentUser() user: User,
  ): any {
    return this.postService.addImage(file, user);
  }
  /**
   *
   * @param selectAllPostsDto
   * @returns all posts
   */
  @HttpCode(200)
  @ApiOkResponse({
    type: GetPostsResponse,
  })
  @Get('/allposts')
  getAllPosts(@Query() selectAllPostsDto: SelectAllPostsDto) {
    return this.postService.getAllPosts(
      selectAllPostsDto.type,
      selectAllPostsDto.priority,
      selectAllPostsDto.take && +selectAllPostsDto.take,
      selectAllPostsDto.skip && +selectAllPostsDto.skip,
    );
  }
  /**
   *
   * @param user
   * @param bookmarkDto
   * @returns Bookmark a post for user
   */
  @ApiBearerAuth('JWT-auth')
  @UseGuards(AuthGuard('jwt'))
  @HttpCode(200)
  @ApiOkResponse({
    type: Bookmarks,
  })
  @Post('/bookmark')
  bookmarkPost(@CurrentUser() user: User, @Body() bookmarkDto: BookmarkDto) {
    return this.postService.bookmarkPost(user, bookmarkDto);
  }
  /**
   *
   * @param user
   * @param paginationDto
   * @returns All current user bookmarks
   */
  @ApiBearerAuth('JWT-auth')
  @UseGuards(AuthGuard('jwt'))
  @HttpCode(200)
  @ApiOkResponse({
    type: GetBookmarksResponse,
  })
  @Get('/getBookmarks')
  getBookmarks(
    @CurrentUser() user: User,
    @Query() paginationDto: PaginationDto,
  ) {
    return this.postService.getBookmarks(
      user,
      paginationDto.take && +paginationDto.take,
      paginationDto.skip && +paginationDto.skip,
    );
  }
  /**
   *
   * @param user
   * @param bookmarkDto
   * @returns a deleted bookmark object
   */
  @ApiBearerAuth('JWT-auth')
  @UseGuards(AuthGuard('jwt'))
  @HttpCode(200)
  @ApiOkResponse({
    type: Bookmarks,
  })
  @Delete('/bookmark')
  deleteBookmark(@CurrentUser() user: User, @Body() bookmarkDto: BookmarkDto) {
    return this.postService.deleteBookmark(user, bookmarkDto);
  }
  /**
   *
   * @param filterDto
   * @returns an array of Filtred posts
   */
  @HttpCode(200)
  @ApiOkResponse({
    type: GetPostsResponse,
  })
  @Get('filtered')
  async getFilteredPosts(
    @Query() searchQuery: SearchQueryDto,
    @Query() filterDto: PostFilterDto,
    @Query() paginationDto: PaginationDto,
  ) {
    return await this.postService.findFilteredPosts(
      searchQuery,
      filterDto,
      paginationDto.take && +paginationDto.take,
      paginationDto.skip && +paginationDto.skip,
    );
  }
  @ApiBearerAuth('JWT-auth')
  @UseGuards(AuthGuard('jwt'))
  @HttpCode(200)
  @ApiOkResponse({
    type: PostClass,
  })
  /***
   * Return a post by id
   */
  @Get(':id')
  async getPost(@Param() getPostDto: GetPostDto) {
    return this.postService.findOne(getPostDto.id);
  }

  @ApiBearerAuth('JWT-auth')
  @UseGuards(AuthGuard('jwt'))
  @HttpCode(200)
  @ApiOkResponse({
    type: PostClass,
  })
  @Patch(':id')
  update(
    @Param() getPostDto: GetPostDto,
    @Body() updatePostDto: UpdatePostDto,
    @CurrentUser() user: User,
  ) {
    return this.postService.update(user, getPostDto.id, updatePostDto);
  }

  @ApiBearerAuth('JWT-auth')
  @UseGuards(AuthGuard('jwt'))
  @HttpCode(200)
  @ApiOkResponse({
    type: PostClass,
  })
  @Delete(':id')
  remove(@Param() getPostDto: GetPostDto, @CurrentUser() user: User) {
    return this.postService.remove(user, getPostDto.id);
  }
}
