import {
  Controller,
  Post,
  Body,
  Req,
  RawBodyRequest,
  HttpCode,
  UseGuards,
  Get,
} from '@nestjs/common';
import { PayementService } from './payement.service';
import { CreatePayementDto } from './dto/create-payement.dto';
import { Request } from 'express';
import { CurrentUser } from '../auth/decorators/current-user.decorator';
import { User } from '../_gen/prisma-class/user';
import { AuthGuard } from '@nestjs/passport';
import { ApiBearerAuth, ApiOkResponse, ApiTags } from '@nestjs/swagger';
import { CheckoutSession } from './entities/payement.entity';
import { ProductListResponseDto } from './entities/productRes';
@ApiTags('payement')
@Controller('payement')
export class PayementController {
  constructor(private readonly payementService: PayementService) {}
  @ApiBearerAuth('JWT-auth')
  @UseGuards(AuthGuard('jwt'))
  @HttpCode(200)
  @ApiOkResponse({
    type: CheckoutSession,
  })
  @Post()
  create(
    @CurrentUser() user: User,
    @Body() createPayementDto: CreatePayementDto,
  ) {
    return this.payementService.create(createPayementDto, user);
  }

  @Post('webhook')
  webhook(@Req() request: RawBodyRequest<Request>) {
    return this.payementService.webhook(request);
  }

  @ApiBearerAuth('JWT-auth')
  @UseGuards(AuthGuard('jwt'))
  @HttpCode(200)
  @ApiOkResponse({
    type: ProductListResponseDto,
  })
  @Get('products')
  getProducts() {
    return this.payementService.getProducts();
  }
}
