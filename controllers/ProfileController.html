import {
  Controller,
  Post,
  UploadedFile,
  UseGuards,
  Get,
  Body,
} from '@nestjs/common';
import { ProfileService } from './profile.service';
import {
  ApiBadRequestResponse,
  ApiBearerAuth,
  ApiOkResponse,
  ApiTags,
  ApiUnauthorizedResponse,
} from '@nestjs/swagger';
import { AuthGuard } from '@nestjs/passport';
import { CurrentUser } from '../auth/decorators/current-user.decorator';
import { ApiImageFile, ApiPdfFile } from './decorators/api-file-decorator';
import { Resume } from '../_gen/prisma-class/resume';
import { Image } from '../_gen/prisma-class/image';
import { CreateActivityDto } from './dto/create-activity.dto';
import { CreateTagsDto } from './dto/create-tags.dto';
import {
  Delete,
  HttpCode,
  Query,
  UploadedFiles,
  UseFilters,
} from '@nestjs/common/decorators';
import { CreateBioDto } from './dto/create-bio.dto';
import { CreateProFocusDto } from './dto/create-profocus.dto';
import { DeleteUserGalleryDto } from './dto/delete-user-gallery.dto';
import { PaginationDto } from './dto/pagination.dto';
import { BookmarkProfileDto } from './dto/bookmark-profile.dto';
import { PrismaClientExceptionFilter } from '../prisma-client-exception/prisma-client-exception.filter';
import { User } from '@prisma/client';
@Controller('profile')
@UseFilters(PrismaClientExceptionFilter)
export class ProfileController {
  constructor(private readonly profileService: ProfileService) {}
  //CREATING PROFILE IMAGE UPLOAD ENDPOINT
  @ApiTags('Profile')
  @Post('uploadProfileImage')
  @ApiImageFile('image')
  uploadProfileImage(
    @UploadedFile()
    file: Express.Multer.File,
    @CurrentUser() user: User,
  ) {
    return this.profileService.uploadProfileImage(file, user);
  }
  //CREATING PROFILE GALLERY UPLOAD ENDPOINT
  @ApiTags('Profile')
  @ApiImageFile('image')
  @Post('uploadGalleryImage')
  uploadGalleryImage(
    @UploadedFile()
    file: Express.Multer.File,
    @CurrentUser() user: User,
  ) {
    return this.profileService.uploadGalleryImage(file, user);
  }
  //CREATING GET GALLERY ENDPOINT
  @ApiTags('Profile')
  @ApiBearerAuth('JWT-auth')
  @Get('gallery')
  @UseGuards(AuthGuard('jwt'))
  @ApiOkResponse({ type: Image, isArray: true })
  getUserGallery(
    @CurrentUser() user: User,
    @Query() paginationDto: PaginationDto,
  ) {
    return this.profileService.getUserGallery(
      user,
      paginationDto.take && +paginationDto.take,
      paginationDto.skip && +paginationDto.skip,
    );
  }
  //CREATING DELETE GALLERY ENDPPINT
  @ApiTags('Profile')
  @ApiBearerAuth('JWT-auth')
  @Delete('gallery')
  @UseGuards(AuthGuard('jwt'))
  @ApiOkResponse({ description: 'Gallery Updated' })
  @ApiUnauthorizedResponse({ description: 'Unauthorized' })
  @ApiBadRequestResponse({ description: 'Bad request payload' })
  @HttpCode(200)
  deleteUserGallery(
    @CurrentUser() user: User,
    @Body() deleteUserGalleryDto: DeleteUserGalleryDto,
  ) {
    return this.profileService.deleteUserGallery(user, deleteUserGalleryDto);
  }
  //CREATING GET GALLERY BY USER ID ENDPONT
  @ApiTags('Profile')
  @ApiBearerAuth('JWT-auth')
  @Get('gallery/:id')
  @UseGuards(AuthGuard('jwt'))
  @ApiOkResponse({ type: Image, isArray: true })
  getUserGalleryById(
    @CurrentUser() user: User,
    @Query('id') id: string,
    @Query() paginationDto: PaginationDto,
  ) {
    return this.profileService.getUserGalleryById(
      user,
      id,
      paginationDto.take && +paginationDto.take,
      paginationDto.skip && +paginationDto.skip,
    );
  }
  //CREATING RESUME UPLOAD ENDPOINT
  @ApiTags('Profile')
  @ApiPdfFile(['document1', 'document2', 'document3'])
  @ApiOkResponse({ type: Resume, isArray: true })
  @Post('uploadResume')
  uploadResume(
    @UploadedFiles() files: Array<Express.Multer.File>,
    @CurrentUser() user: User,
  ) {
    return this.profileService.uploadResume(files, user);
  }
  //ENDPOINT TO GET ALL USER RESUMES
  @ApiTags('Profile')
  @ApiBearerAuth('JWT-auth')
  @Get('resumes')
  @UseGuards(AuthGuard('jwt'))
  @ApiOkResponse({ type: Resume, isArray: true })
  @HttpCode(200)
  getUserResumes(@CurrentUser() user: User) {
    return this.profileService.getUserResumes(user);
  }
  //ENDPOINT TO GET ALL USER RESUMES BY ID
  @ApiTags('Profile')
  @ApiBearerAuth('JWT-auth')
  @Get('resumes/:id')
  @UseGuards(AuthGuard('jwt'))
  @ApiOkResponse({ type: Resume, isArray: true })
  @HttpCode(200)
  getUserResumeById(@CurrentUser() user: User, @Query('id') id: string) {
    return this.profileService.getUserResumeById(user, id);
  }
  //ENDPOINT TO SET USER ACTIVITY
  @ApiTags('Profile')
  @ApiBearerAuth('JWT-auth')
  @Post('activity')
  @UseGuards(AuthGuard('jwt'))
  @ApiOkResponse({ description: 'User activity set' })
  @ApiUnauthorizedResponse({ description: 'Unauthorized' })
  @ApiBadRequestResponse({ description: 'Bad request payload' })
  @HttpCode(200)
  setUserActivity(
    @CurrentUser() user: User,
    @Body() createActivityDto: CreateActivityDto,
  ) {
    return this.profileService.setUserActivity(user, createActivityDto);
  }

  //ENDPOINT TO SET USER TAGS
  @ApiTags('Profile')
  @ApiBearerAuth('JWT-auth')
  @Post('tags')
  @UseGuards(AuthGuard('jwt'))
  @ApiOkResponse({ description: 'User Tags set' })
  @ApiUnauthorizedResponse({ description: 'Unauthorized' })
  @ApiBadRequestResponse({ description: 'Bad request payload' })
  @HttpCode(200)
  SetUserTags(@CurrentUser() user: User, @Body() createTagsDto: CreateTagsDto) {
    return this.profileService.setUserTags(user, createTagsDto);
  }
  //ENDPOINT TO SET THE BIO
  @ApiTags('Profile')
  @ApiBearerAuth('JWT-auth')
  @Post('bio')
  @UseGuards(AuthGuard('jwt'))
  setBio(@CurrentUser() user: User, @Body() createBioDto: CreateBioDto) {
    return this.profileService.setUserBio(user, createBioDto);
  }
  //ENDPOINT TO SET USER PROFOCUS
  @ApiTags('Profile')
  @ApiBearerAuth('JWT-auth')
  @Post('proFocus')
  @UseGuards(AuthGuard('jwt'))
  setProFocus(
    @CurrentUser() user: User,
    @Body() createProFocusDto: CreateProFocusDto,
  ) {
    return this.profileService.setProFocus(user, createProFocusDto);
  }
  //ENDPOINT TO GET USER PROFOCUS
  @ApiTags('Profile')
  @ApiBearerAuth('JWT-auth')
  @ApiOkResponse({
    type: String,
    isArray: true,
  })
  @Get('proFocus')
  @UseGuards(AuthGuard('jwt'))
  getProFocus(@CurrentUser() user: User) {
    return this.profileService.getProFocus(user);
  }
  //ENDPOINT TO DELETE USER
  @ApiTags('User Settings')
  @ApiBearerAuth('JWT-auth')
  @Delete('deleteUser')
  @UseGuards(AuthGuard('jwt'))
  @ApiOkResponse({ description: 'User Deleted' })
  @ApiUnauthorizedResponse({ description: 'Unauthorized' })
  @ApiBadRequestResponse({ description: 'Bad request payload' })
  @HttpCode(200)
  deleteUser(@CurrentUser() user: User) {
    return this.profileService.deleteUser(user);
  }

  //ENDPOINT TO BOOKMARK USER PROFILE
  @ApiTags('Profile')
  @ApiBearerAuth('JWT-auth')
  @Post('bookmarkProfile')
  @UseGuards(AuthGuard('jwt'))
  @ApiOkResponse({ description: 'Profile Bookmarked' })
  @ApiUnauthorizedResponse({ description: 'Unauthorized' })
  @ApiBadRequestResponse({ description: 'Bad request payload' })
  @HttpCode(200)
  bookmarkProfile(
    @CurrentUser() user: User,
    @Body() bookmarkProfileDto: BookmarkProfileDto,
  ) {
    return this.profileService.bookmarkProfile(user, bookmarkProfileDto);
  }

  //ENDPOINT TO GET BOOKMARKED PROFILES
  @ApiTags('Profile')
  @ApiBearerAuth('JWT-auth')
  @Get('bookmarkedProfiles')
  @UseGuards(AuthGuard('jwt'))
  @ApiOkResponse({ description: 'Profile Bookmarked' })
  @ApiUnauthorizedResponse({ description: 'Unauthorized' })
  @ApiBadRequestResponse({ description: 'Bad request payload' })
  @HttpCode(200)
  getBookmarkedProfiles(
    @CurrentUser() user: User,
    @Query() paginationDto: PaginationDto,
  ) {
    return this.profileService.getBookmarkedProfiles(
      user,
      paginationDto.take && +paginationDto.take,
      paginationDto.skip && +paginationDto.skip,
    );
  }
}
