import {
  Controller,
  Post,
  Get,
  Body,
  Param,
  UseFilters,
  UseGuards,
  HttpCode,
  HttpStatus,
  BadRequestException,
  Patch,
} from '@nestjs/common';
import { AuthService } from './auth.service';
import { CreateUserDto } from './dto/create-user.dto';
import { CreateAuthDto } from './dto/create-auth.dto';
import { AuthGuard } from '@nestjs/passport';
import { Tokens } from './types';
import { UserFoundGuard } from './guards/user-found.guard';
import { PrismaClientExceptionFilter } from '../prisma-client-exception/prisma-client-exception.filter';
import { UpdatePwdDto } from './dto/update-pwd.dto';
import { ResetPwdDto } from './dto/reset-pwd.dto';
import {
  ApiBadRequestResponse,
  ApiBearerAuth,
  ApiConflictResponse,
  ApiCreatedResponse,
  ApiNotFoundResponse,
  ApiOkResponse,
  ApiTags,
  ApiUnauthorizedResponse,
} from '@nestjs/swagger';
import { CurrentUser } from './decorators/current-user.decorator';
import { User as UserClass } from '../_gen/prisma-class/user';
import { User } from '@prisma/client';
import { GetUserDto } from './dto/get-user.dto';
import { SetUserSettingsDto } from './dto/update-user-settings.dto';
import { ChangePasswordDto } from './dto/change-pwd.dto';
import { ChangeEmailDto } from './dto/change-email.dto';
import { HandleChangeEmailDto } from './dto/handle-change-email.dto';

@Controller('auth')
@UseFilters(PrismaClientExceptionFilter)
export class AuthController {
  constructor(private readonly authService: AuthService) {}
  //ENDPOINT TO REGISTER A USER
  @ApiTags('Register')
  @ApiCreatedResponse({ description: 'User created' })
  @ApiBadRequestResponse({
    description: 'Check the error message: Error in validating your json',
  })
  @ApiConflictResponse({
    description: 'User with this data exist',
  })
  @UseGuards(UserFoundGuard)
  @Post('register')
  create(@Body() createUserDto: CreateUserDto) {
    return this.authService.create(createUserDto);
  }
  //ENDPOINT TO VERIFY USER EMAIL
  @HttpCode(HttpStatus.OK)
  @ApiOkResponse({
    description: 'User verified',
  })
  @ApiNotFoundResponse({
    description: 'Token not found',
  })
  @ApiTags('Register')
  @Get('verify/:token')
  verify(@Param('token') token: string) {
    return this.authService.verifyUser(token);
  }
  //ENDPOINT TO LOGIN THE USER
  @ApiOkResponse({
    description: 'User logged in',
  })
  @ApiUnauthorizedResponse({
    description: 'Invalid credentials',
  })
  @ApiTags('Login')
  @Post('login')
  login(@Body() createAuthDto: CreateAuthDto) {
    return this.authService.login(createAuthDto);
  }
  //ENDPOINT TO REMOVE THE TOKENS AND LOGOUT THE USER
  @ApiTags('Login')
  @UseGuards(AuthGuard('jwt'))
  @Post('logout')
  logout(@CurrentUser() user: User): Promise<boolean> {
    const userId = user.id;
    return this.authService.logout(userId);
  }
  //ENDPOINT TO REFRESH TOKENS
  @ApiTags('Login')
  @ApiBearerAuth('JWT-auth')
  @UseGuards(AuthGuard('jwt-refresh'))
  @Post('refresh')
  refreshTokens(@CurrentUser() user: UserClass): Promise<Tokens> {
    const userId = user.id;
    if (user.refreshToken)
      return this.authService.refreshTokens(
        userId,
        user.refreshToken.toString(),
      );
    throw new BadRequestException('Not Refresh token provided ');
  }
  //ENDPOINT TO SEND PASSWORD RESET LINK
  @HttpCode(HttpStatus.OK)
  @ApiTags('Reset Password')
  @Post('reset')
  generateResetPwd(@Body() resetPwdDto: ResetPwdDto) {
    return this.authService.generateResetPwd(resetPwdDto.email);
  }
  //ENDPOINT TO RESET (CHANGE) THE PASSWORD BASED ON TOKEN SENT IN URL AND NEW PASSWORD SENT IN BODY
  @HttpCode(HttpStatus.OK)
  @ApiTags('Reset Password')
  @Post('reset/:token')
  resetPwd(@Param('token') token: string, @Body() updatePwdDto: UpdatePwdDto) {
    return this.authService.resetPwd(token, updatePwdDto);
  }
  //ENDPOINT TO GET THE CURRENT USER BASED ON JWT SENT IN HEADER
  @UseGuards(AuthGuard('jwt'))
  @HttpCode(HttpStatus.OK)
  @ApiUnauthorizedResponse({
    description: 'Not logged in',
  })
  @ApiOkResponse({
    type: UserClass,
  })
  @ApiBearerAuth('JWT-auth')
  @ApiTags('User')
  @Get('/getuser')
  getUser(@CurrentUser() user: User) {
    return this.authService.getUser(user);
  }
  //ENDPOINT TO GET USER USING ID
  @UseGuards(AuthGuard('jwt'))
  @HttpCode(HttpStatus.OK)
  @ApiUnauthorizedResponse({
    description: 'Not logged in',
  })
  @ApiOkResponse({
    type: UserClass,
  })
  @ApiBearerAuth('JWT-auth')
  @ApiTags('User')
  @Get('/getuser/:id')
  getUserById(@CurrentUser() user: User, @Param() getUserDto: GetUserDto) {
    return this.authService.getUserById(user, getUserDto.id);
  }

  //ENDPOINT TO GET USER INFO
  @ApiTags('User Settings')
  @ApiBearerAuth('JWT-auth')
  @ApiOkResponse({ description: 'User Settings set' })
  @ApiUnauthorizedResponse({ description: 'Unauthorized' })
  @ApiBadRequestResponse({ description: 'Bad request payload' })
  @HttpCode(200)
  @UseGuards(AuthGuard('jwt'))
  @Patch('settings')
  updateSettings(
    @CurrentUser() user: User,
    @Body() setUserSettingsDto: SetUserSettingsDto,
  ) {
    return this.authService.updateSettings(user, setUserSettingsDto);
  }

  //ENPOINT TO CHANGE USER PASSWORD BASED ON OLD PASSWORD
  @ApiTags('User Settings')
  @ApiBearerAuth('JWT-auth')
  @ApiOkResponse({ type: UserClass })
  @ApiUnauthorizedResponse({
    description: 'Unauthorized / Old password is incorrect',
  })
  @ApiBadRequestResponse({ description: 'Bad request payload' })
  @ApiNotFoundResponse({ description: 'User not found' })
  @HttpCode(200)
  @UseGuards(AuthGuard('jwt'))
  @Patch('changepwd')
  changePwd(
    @CurrentUser() user: User,
    @Body() updatePwdDto: ChangePasswordDto,
  ): Promise<User> {
    return this.authService.ChangePassword(user, updatePwdDto);
  }
  //ENPOINT TO CHANGE USER EMAIL
  @ApiTags('User Settings')
  @ApiBearerAuth('JWT-auth')
  @ApiUnauthorizedResponse({
    description: 'Unauthorized / Old password is incorrect',
  })
  @ApiBadRequestResponse({ description: 'Bad request payload' })
  @ApiNotFoundResponse({ description: 'User not found' })
  @HttpCode(200)
  @UseGuards(AuthGuard('jwt'))
  @Patch('changeEmail')
  changeEmail(
    @CurrentUser() user: User,
    @Body() changeEmailDto: ChangeEmailDto,
  ): Promise<{ message: string } | undefined> {
    return this.authService.changeEmail(user, changeEmailDto.password, changeEmailDto.newEmail);
  }
  //ENPOINT TO HANDLE CHANGE USER EMAIL
  @ApiTags('User Settings')
  @ApiOkResponse({ description: 'new tokens' })
  @ApiUnauthorizedResponse({
    description: 'Unauthorized',
  })
  @ApiBadRequestResponse({ description: 'Bad request payload' })
  @ApiNotFoundResponse({ description: 'User not found' })
  @HttpCode(200)
  @Get('changeEmail/:token')
  handleChangeEmail(
    @Param() tokenDto: HandleChangeEmailDto,
  ): Promise<Tokens> {
    return this.authService.handleChangeEmail(tokenDto.token);
  }
}
