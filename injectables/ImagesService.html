import { HttpException, Injectable } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';
import { FilerService } from '../digitalocean-filer/digitalocean-filer.service';
import { Image, User } from '@prisma/client';

@Injectable()
export class ImagesService {
  constructor(
    private readonly prisma: PrismaService,
    private readonly filerService: FilerService,
  ) {}
  async createImage(
    file: Express.Multer.File,
    user: User,
    isUserProfilePicture: boolean,
    isGalleryImage?: boolean,
    isPostImage?: boolean,
  ): Promise<Image> {
    //CHECKING FOR FILE SIZE
    if (!file) {
      throw new HttpException('No file provided', 413);
    }
    if (file.size > 4242880) {
      throw new HttpException('File too large', 413);
    }
    //RETURN IMAGE LINK
    const imageLinkAfterUpload = await this.filerService.uploadFile(file);

    //CREATE IMAGE INSTANCE IN DB
    const image = await this.prisma.image.create({
      data: {
        imageUrl: imageLinkAfterUpload,
        isUserProfilePicture: isUserProfilePicture,
        ownerId: user.id,
        isGalleryImage,
        isPostImage,
      },
    });
    return image;
  }
  async getUserGallery(
    user: User,
    take?: number,
    skip?: number,
  ): Promise<Image[]> {
    if ((skip && isNaN(skip)) || (take && isNaN(take))) {
      return this.prisma.image.findMany({
        where: {
          ownerId: user.id,
          isUserProfilePicture: false,
          isGalleryImage: true,
        },
        orderBy: {
          createdAt: 'desc',
        },
      });
    } else {
      return await this.prisma.image.findMany({
        take,
        skip,
        where: {
          ownerId: user.id,
          isUserProfilePicture: false,
          isGalleryImage: true,
        },
        orderBy: {
          createdAt: 'desc',
        },
      });
    }
  }
}
