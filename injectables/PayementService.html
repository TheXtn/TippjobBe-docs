import { HttpException, Injectable, RawBodyRequest } from '@nestjs/common';
import { CreatePayementDto } from './dto/create-payement.dto';
import Stripe from 'stripe';
import { Request } from 'express';
import { User } from '../_gen/prisma-class/user';
import { PrismaService } from '../prisma/prisma.service';
import { addHours } from 'date-fns';
@Injectable()
export class PayementService {
  private stripe;

  constructor(private readonly prisma: PrismaService) {
    this.stripe = new Stripe(process.env.STRIPE_KEY || '', {
      apiVersion: '2022-11-15',
    });
  }

  /**
   *
   * @param createPayementDto
   * @param user
   * @returns Payment session : generated url for payment that redirect to the front end
   */
  async create(createPayementDto: CreatePayementDto, user: User) {
    const userEmail = await this.prisma.user.findUnique({
      where: {
        id: user.id,
      },
      select: {
        email: true,
      },
    });
    /**
     * Creating a session ( including creation of stripe account based on the user email)
     */
    const item = await this.stripe.prices.retrieve(
      createPayementDto.productPriceId,
    );
    if (item.unit_amount == 0) {
      await this.updatePostPayement(
        createPayementDto.postId,
        createPayementDto.productPriceId,
        user.id,
      );
      throw new HttpException(
        'Amount is 0 so its updated without payement',
        201,
      );
    }
    const session = await this.stripe.checkout.sessions.create({
      customer_email: userEmail?.email,
      line_items: [
        {
          // Provide the exact Price ID (for example, pr_1234) of the product you want to sell
          price: createPayementDto.productPriceId,
          quantity: 1,
        },
      ],
      metadata: {
        userId: user.id,
        postId: createPayementDto.postId,
        productId: createPayementDto.productPriceId,
      },
      mode: 'payment',
      payment_method_types: ['card'],
      success_url:
        `${process.env.STRIPE_SUCCESS_PAYEMENT_URL}?tab=advert&payment=success` ||
        '',
      cancel_url: `${process.env.STRIPE_SUCCESS_PAYEMENT_URL}?tab=advert&payment=error`,
    });
    console.log(session);
    return session;
  }
  /**
   * This function triggered on the webhook of success payment
   * @param postId
   * @param productId
   * @param userId
   * @returns An updated post object
   */
  async updatePostPayement(
    postId?: string,
    productId?: string,
    userId?: string,
  ) {
    const price = (await this.stripe.prices.retrieve(productId || '', {
      expand: ['product'],
    })) as Stripe.Price & { product: Stripe.Product };
    console.log('Postid=', postId);
    console.log('Period=', +price.product.metadata['period']);
    const post = await this.prisma.post.findUnique({
      where: {
        id: postId,
      },
    });
    const newExpirationDate = addHours(
      new Date(),
      +price.product.metadata['period'],
    );
    const updatedPost = await this.prisma.post.updateMany({
      where: {
        id: post?.id,
        userId,
      },
      data: {
        expirationDate: newExpirationDate,
      },
    });
    return updatedPost;
  }
  /**
   *
   * @param request
   * @returns Nothing it justs the webhook endpoint that trigger on success payment
   */
  async webhook(request: RawBodyRequest<Request>) {
    // checkout-session.type.ts

    type CheckoutSession = {
      id: string;
      object: 'checkout.session';
      billing_address_collection?: 'auto' | 'required';
      cancel_url: string;
      client_reference_id?: string;
      customer?: string;
      customer_email?: string;
      line_items: Stripe.LineItem[];
      locale?: string;
      mode: 'payment' | 'setup' | 'subscription' | 'subscription_update';
      payment_intent?: string;
      payment_method_types: string[];
      payment_status?: 'paid' | 'unpaid';
      setup_intent?: string;
      shipping_address_collection?: {
        allowed_countries: string[];
      };
      submit_type?: 'auto' | 'book' | 'donate' | 'pay';
      subscription?: string;
      success_url: string;
      metadata?: {
        [key: string]: string;
      };
    };

    const endpointSecret = process.env.STRIPE_WEBHOOK_SECRET || '';
    let event;

    try {
      const sig = request.headers['stripe-signature'] || '';
      const rawBody = request.rawBody || ''; // Assuming you have access to the raw request body
      event = this.stripe.webhooks.constructEvent(rawBody, sig, endpointSecret);
    } catch (e) {}
    switch (event?.type) {
      case 'checkout.session.completed':
        const checkoutSession = event.data.object as CheckoutSession;
        if (checkoutSession.payment_status == 'paid') {
          await this.updatePostPayement(
            checkoutSession.metadata && checkoutSession.metadata['postId'],
            checkoutSession.metadata && checkoutSession.metadata['productId'],
            checkoutSession.metadata && checkoutSession.metadata['userId'],
          );
        }
        break;
      default:
        console.log(`Unhandled event type ${event?.type}`);
    }
    return `This action returns nothing`;
  }
  /**
   *
   * @returns all products in stripe
   */
  async getProducts() {
    return this.stripe.products.list({
      active: true,
    });
  }
}
