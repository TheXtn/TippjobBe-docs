import { Inject, Injectable } from '@nestjs/common';
import * as AWS from 'aws-sdk';
import {
  DoSpacesServiceLib,
  UploadedMulterFileI,
} from './digitalocean-filer.config';
@Injectable()
export class FilerService {
  constructor(@Inject(DoSpacesServiceLib) private readonly s3: AWS.S3) {}

  async uploadFile(
    file: UploadedMulterFileI,
    downloadable?: boolean,
  ): Promise<string> {
    // Precaution to avoid having 2 files with the same name
    const fileName = `${Date.now()}-${file.originalname}`;

    // Return a promise that resolves only when the file upload is complete
    return new Promise((resolve, reject) => {
      const params: AWS.S3.PutObjectRequest = {
        Bucket: process.env.DIGITAL_OCEAN_SPACES_BUCKET || '',
        Key: 'TIPPJOB/' + fileName,
        Body: file.buffer,
        ACL: 'public-read',
      };

      if (!downloadable) {
        params.ContentType = file.mimetype;
      }

      this.s3.putObject(params, (error: AWS.AWSError) => {
        if (!error) {
          resolve(
            `${process.env.DIGITAL_OCEAN_SPACES_INSTANCE_LINK}/${fileName}`,
          );
        } else {
          reject(new Error(`DoSpacesService_ERROR: ${error}`));
        }
      });
    });
  }
}
