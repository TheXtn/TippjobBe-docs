import {
  BadRequestException,
  Injectable,
  UnauthorizedException,
} from '@nestjs/common';
import { User, Image, Resume } from '@prisma/client';
import { PrismaService } from '../prisma/prisma.service';
import { ImagesService } from '../images/images.service';
import { ResumesService } from '../resumes/resumes.service';
import { CreateActivityDto } from './dto/create-activity.dto';
import { CreateTagsDto } from './dto/create-tags.dto';
import { ActivitiesService } from '../activities/activities.service';
import { CreateBioDto } from './dto/create-bio.dto';
import { CreateProFocusDto } from './dto/create-profocus.dto';
import { DeleteUserGalleryDto } from './dto/delete-user-gallery.dto';
import { BookmarkProfileDto } from './dto/bookmark-profile.dto';
@Injectable()
export class ProfileService {
  constructor(
    private readonly prisma: PrismaService,
    private readonly imagesService: ImagesService,
    private readonly resumeService: ResumesService,
    private readonly activityService: ActivitiesService,
  ) {}
  //SERVICE TO UPLOAD USER PROFILE IMAGE
  async uploadProfileImage(file: Express.Multer.File, user: User) {
    const image = await this.imagesService.createImage(file, user, true);
    //CREATE A PROFILE IF NOT EXIST | UPDATE EXISTING PROFILE -> SET NEW PROFILE IMAGE
    const profile = await this.prisma.profile.upsert({
      where: {
        userId: user.id,
      },
      update: {
        profileImage: image.imageUrl,
      },
      create: {
        user: {
          connect: {
            id: user.id,
          },
        },

        profileImage: image.imageUrl,
      },
      include: {
        user: true,
      },
    });
    return profile;
  }
  //SERVICE TO UPLOAD AN IMAGE TO USER IMAGES GALLERY
  async uploadGalleryImage(
    file: Express.Multer.File,
    user: User,
  ): Promise<Image> {
    return await this.imagesService.createImage(file, user, false, true);
  }
  //SERVICE TO GET ALL USER IMAGES GALLERY
  async getUserGallery(
    user: User,
    take?: number,
    skip?: number,
  ): Promise<Image[]> {
    return await this.imagesService.getUserGallery(user, take, skip);
  }
  //SERVICE TO DELETE USER IMAGES GALLERY
  async deleteUserGallery(
    user: User,
    deleteUserGalleryDto: DeleteUserGalleryDto,
  ) {
    await this.prisma.image.deleteMany({
      where: {
        ownerId: user.id,
        id: {
          in: deleteUserGalleryDto.idsArray,
        },
      },
    });
    return 'Gallery Updated';
  }
  //SERVICE TO GET USER GALLERY BY USER ID
  async getUserGalleryById(
    user: User,
    id: string,
    take?: number,
    skip?: number,
  ): Promise<Image[] | null> {
    if (!id) {
      return null;
    }
    //GET THE CURRENT USER (JWT USER)
    const currentUser = await this.prisma.user.findUnique({
      where: { id: user.id },
    });
    //GET THE USER BASE ON THE ID PATH PARAM
    const dbUser = await this.prisma.user.findUnique({
      where: {
        id: id,
      },
      include: {
        profile: {
          include: {
            activity: true,
          },
        },
      },
    });
    if (dbUser) {
      if (dbUser.visibility == 'PUBLIC') {
        return await this.imagesService.getUserGallery(dbUser, take, skip);
      }
      if (
        dbUser.visibility == 'COMPANY' &&
        currentUser?.userType == 'COMPANY'
      ) {
        return await this.imagesService.getUserGallery(dbUser, take, skip);
      }
    }
    return null;
  }
  // THIS FUNCTION TAKES IN AN ARRAY OF FILES AND A USER OBJECT, AND RETURNS AN ARRAY OF UPLOADED RESUMES.
  async uploadResume(
    files: Array<Express.Multer.File>,
    user: User,
  ): Promise<Resume[]> {
    // INITIALIZE AN EMPTY ARRAY TO STORE UPLOADED RESUMES.
    const arrayOfUploadedResumes: Resume[] = [];

    // LOOP THROUGH EACH FILE IN THE ARRAY OF FILES.
    for (const file of files) {
      // CHECK IF THE FILE IS A PDF BY VERIFYING ITS MIMETYPE.
      if (file.mimetype != 'application/pdf') {
        // THROW A BADREQUESTEXCEPTION IF THE FILE IS NOT A PDF.
        throw new BadRequestException(`Check your file ${file.originalname}`);
      }

      // CHECK IF A RESUME WITH THE SAME FILE NAME ALREADY EXISTS FOR THE USER.
      const res = await this.prisma.resume.findFirst({
        where: {
          ownerId: user.id,
          resumeUrl:
            'https://cc-dev-storage.fra1.digitaloceanspaces.com/TIPPJOB/' +
            file.originalname,
        },
      });

      // IF A RESUME WITH THE SAME FILE NAME DOES NOT EXIST, CREATE A NEW RESUME AND ADD IT TO THE ARRAY OF UPLOADED RESUMES.
      if (!res) {
        const resume = await this.resumeService.createResume(file, user);
        arrayOfUploadedResumes.push(resume);
      }
      // IF A RESUME WITH THE SAME FILE NAME EXISTS, ADD IT TO THE ARRAY OF UPLOADED RESUMES.
      else {
        arrayOfUploadedResumes.push(res);
      }
    }

    // DELETE ANY EXISTING RESUMES FOR THE USER THAT WERE NOT UPLOADED IN THIS BATCH.
    await this.resumeService.deleteResumes(
      arrayOfUploadedResumes.map((resume) => resume.id),
      user.id,
    );

    // RETURN THE ARRAY OF UPLOADED RESUMES.
    return arrayOfUploadedResumes;
  }

  //SERVICE TO GET ALL USER RESUMES
  async getUserResumes(user: User): Promise<Resume[]> {
    return await this.resumeService.getUserResumes(user);
  }

  //SERVICE TO GET ALL USER RESUMES BY ID
  async getUserResumeById(user: User, id: string): Promise<Resume[] | null> {
    if (!id) {
      return null;
    }
    //GET THE CURRENT USER (JWT USER)
    const currentUser = await this.prisma.user.findUnique({
      where: { id: user.id },
    });
    if (!currentUser) {
      throw new UnauthorizedException('Unauthorized');
    }
    //GET THE USER BASE ON THE ID PATH PARAM
    const dbUser = await this.prisma.user.findUnique({
      where: {
        id: id,
      },
      include: {
        profile: {
          include: {
            activity: true,
          },
        },
      },
    });
    if (dbUser) {
      if (dbUser.uploadsVisibility == 'PUBLIC') {
        return await this.resumeService.getUserResumes(dbUser);
      }
      if (
        dbUser.uploadsVisibility == 'COMPANY' &&
        currentUser?.userType == 'COMPANY'
      ) {
        return await this.resumeService.getUserResumes(dbUser);
      }
    }
    return null;
  }

  //SERVICE TO SET USER ACTIVITY
  async setUserActivity(user: User, CreateActivityDto: CreateActivityDto) {
    const profile = await this.prisma.profile.findUnique({
      where: {
        userId: user.id,
      },
    });
    if (profile) {
      return await this.activityService.upsertActivity(
        profile,
        CreateActivityDto,
      );
    } else {
      return await this.activityService.createActivity(user, CreateActivityDto);
    }
  }
  //SERVICE TO SET USER BIO
  async setUserBio(user: User, createBioDto: CreateBioDto) {
    return await this.prisma.profile.upsert({
      where: {
        userId: user.id,
      },
      create: {
        bio: createBioDto.bio,
        user: {
          connect: {
            id: user.id,
          },
        },
      },
      update: {
        bio: createBioDto.bio,
      },
    });
  }

  //SERVICE TO SET USER TAGS
  async setUserTags(user: User, body: CreateTagsDto) {
    const values = Object.values(body);
    if (new Set(values).size !== values.length) {
      throw new BadRequestException('Body contains duplicate values');
    }

    body.tag_1 = body.tag_1 || null;
    body.tag_2 = body.tag_2 || null;
    body.tag_3 = body.tag_3 || null;

    await this.prisma.profile.upsert({
      where: {
        userId: user.id,
      },
      update: {
        tag_1: body.tag_1,
        tag_2: body.tag_2,
        tag_3: body.tag_3,
      },
      create: {
        user: {
          connect: {
            id: user.id,
          },
        },
        tag_1: body.tag_1,
        tag_2: body.tag_2,
        tag_3: body.tag_3,
      },
    });
    return { message: 'Tag(s) set successfully' };
  }
  //SERVICE TO SET USER PROFOCUS
  async setProFocus(
    user: User,
    createProFocus: CreateProFocusDto,
  ): Promise<string[] | undefined> {
    const profile = await this.prisma.profile.upsert({
      where: {
        userId: user.id,
      },
      create: {
        proFocus: createProFocus.proFocusArray.join(','),
        user: {
          connect: {
            id: user.id,
          },
        },
      },
      update: {
        proFocus: createProFocus.proFocusArray.join(','),
      },
    });
    return profile?.proFocus?.split(',');
  }
  async getProFocus(user: User): Promise<string[]> {
    const profile = await this.prisma.profile.findFirst({
      where: {
        userId: user.id,
      },
    });
    if (profile?.proFocus) return profile.proFocus.split(',');
    else return [];
  }
  //SERVICE TO SET THE USER AS DELETED
  async deleteUser(user: User) {
    //delete the user compeletely
    await this.prisma.user.delete({
      where: {
        id: user.id,
      },
    });
    return { message: 'User Deleted' };
  }

  //SERVICE TO BOOKMARK USER PROFILE
  async bookmarkProfile(user: User, bookmarkProfileDto: BookmarkProfileDto) {
    const mark = await this.prisma.profileBookmarks.findFirst({
      where: {
        bookmarkedUserId: bookmarkProfileDto.bookmarkedUserId,
        bookmarkingUserId: user.id,
      },
    });
    if (mark) {
      await this.prisma.profileBookmarks.delete({
        where: {
          bookmarkingUserId_bookmarkedUserId: {
            bookmarkingUserId: user.id,
            bookmarkedUserId: bookmarkProfileDto.bookmarkedUserId,
          },
        },
      });
      return { message: 'Profile Unbookmarked' };
    }

    await this.prisma.profileBookmarks.create({
      data: {
        bookmarkingUserId: user.id,
        bookmarkedUserId: bookmarkProfileDto.bookmarkedUserId,
      },
    });
    return { message: 'Profile Bookmarked' };
  }

  //SERVICE TO GET BOOKMARKED PROFILES
  async getBookmarkedProfiles(user: User, take?: number, skip?: number) {
    const [bookmarks, totalCount] = await Promise.all([
      this.prisma.profileBookmarks.findMany({
        where: {
          bookmarkingUserId: user.id,
        },
        include: {
          bookmarkedUser: {
            include: {
              profile: true,
            },
          },
        },
        take,
        skip,
      }),

      this.prisma.profileBookmarks.count({
        where: {
          bookmarkingUserId: user.id,
        },
      }),
    ]);
    return {
      totalCount,
      bookmarks,
    };
  }
}
