import {
  BadRequestException,
  ForbiddenException,
  Injectable,
} from '@nestjs/common';
import { CreatePostDto } from './dto/create-post.dto';
import { UpdatePostDto } from './dto/update-post.dto';
import { PrismaService } from '../prisma/prisma.service';
import { User, Image, JobPriority, JobType } from '@prisma/client';
import { ImagesService } from '../images/images.service';
import { BookmarkDto } from './dto/bookmark.dto';
import { PostFilterDto } from './dto/post-filter.dto';

import { SearchQueryDto } from './dto/create-search.dto';

@Injectable()
export class PostService {
  constructor(
    private readonly prisma: PrismaService,
    private readonly imagesService: ImagesService,
  ) {}
  async create(createPostDto: CreatePostDto, user: User) {
    return await this.prisma.post.create({
      data: {
        ...createPostDto,
        extra: createPostDto.extra?.join(','),
        user: {
          connect: {
            id: user.id,
          },
        },
      },
    });
  }
  /**
   *
   * @param user
   * @param userId
   * @param type
   * @param priority
   * @param take
   * @param skip
   * @returns All user posts
   */
  async findAllUserPosts(
    user: User,
    userId: string,
    type: JobType,
    priority: JobPriority,
    take?: number,
    skip?: number,
  ) {
    if (userId) {
      const [posts, totalCount] = await Promise.all([
        this.prisma.post.findMany({
          where: {
            userId: userId,
            priority: priority || undefined,
            type: type || undefined,
            expirationDate: {
              gt: new Date().toISOString(),
            },
          },
          take,
          skip,
          include: {
            user: true,
          },
          orderBy: {
            expirationDate: 'desc',
          },
        }),
        this.prisma.post.count({
          where: {
            priority: priority || undefined,
            type: type,
          },
        }),
      ]);
      return {
        totalCount,
        posts,
      };
    } else {
      const [posts, totalCount] = await Promise.all([
        this.prisma.post.findMany({
          where: {
            userId: user.id,
            priority: priority || undefined,
            type: type || undefined,
          },
          take,
          skip,
          include: {
            user: true,
          },
          orderBy: {
            expirationDate: 'desc',
          },
        }),
        this.prisma.post.count({
          where: {
            priority: priority || undefined,
            type: type,
          },
        }),
      ]);
      return {
        totalCount,
        posts,
      };
    }
  }
  async getAllPosts(
    type: JobType,
    priority: JobPriority,
    take?: number,
    skip?: number,
  ) {
    if (type === 'JOB_AD') {
      const [posts, totalCount] = await Promise.all([
        this.prisma.post.findMany({
          where: {
            priority: priority || undefined,
            type: type,
            expirationDate: {
              gt: new Date().toISOString(),
            },
          },
          take,
          skip,
          include: {
            user: true,
          },
        }),
        this.prisma.post.count({
          where: {
            priority: priority || undefined,
            type: type,
            expirationDate: {
              gt: new Date().toISOString(),
            },
          },
        }),
      ]);

      return {
        totalCount,
        posts,
      };
    } else {
      const [posts, totalCount] = await Promise.all([
        this.prisma.post.findMany({
          where: {
            priority: priority || undefined,
            NOT: {
              type: 'JOB_AD',
            },
            expirationDate: {
              gt: new Date().toISOString(),
            },
          },
          take,
          skip,
          include: {
            user: true,
          },
        }),
        this.prisma.post.count({
          where: {
            priority: priority || undefined,
            NOT: {
              type: 'JOB_AD',
            },

            expirationDate: {
              gt: new Date().toISOString(),
            },
          },
        }),
      ]);
      return {
        totalCount,
        posts,
      };
    }
  }
  /**
   * @param file
   * @param user
   * @return an uploaded image of type Image
   */
  async addImage(file: Express.Multer.File, user: User): Promise<Image> {
    return await this.imagesService.createImage(file, user, false, false, true);
  }
  /**
   *
   * @param user
   * @param postId
   * @returns bookmark object that represent a bookmarked post by a user id
   */
  async bookmarkPost(user: User, bookmarkDto: BookmarkDto) {
    return await this.prisma.bookmarks.create({
      data: {
        userId: user.id,
        postId: bookmarkDto.postId,
      },
    });
  }
  /**
   *
   * @param user
   * @returns a list of bookmarked posts of specific user
   */
  async getBookmarks(user: User, take?: number, skip?: number) {
    const [bookmarks, totalCount] = await Promise.all([
      this.prisma.bookmarks.findMany({
        where: {
          userId: user.id,
        },
        select: {
          post: {
            include: {
              user: true,
            },
          },
        },
        take,
        skip,
      }),
      this.prisma.bookmarks.count({
        where: {
          userId: user.id,
        },
      }),
    ]);
    return {
      totalCount,
      bookmarks,
    };
  }
  /**
   *
   * @param user
   * @param postId
   * @returns deleted bookmark object
   */
  async deleteBookmark(user: User, bookmarkDto: BookmarkDto) {
    return await this.prisma.bookmarks.delete({
      where: {
        postId_userId: { postId: bookmarkDto.postId, userId: user.id },
      },
    });
  }
  /**
   * Finds and returns an array of Post objects filtered by the given criteria.
   * @async
   * @function findFilteredPosts
   * @param {PostFilterDto} filterDto - The object containing filter criteria.
   * @returns {Promise<Post[]>} - A promise that resolves to an array of Post objects.
   */

  removeUndefinedValues(obj: any) {
    // recursively iterate through each key-value pair in the object
    for (const prop in obj) {
      if (typeof obj[prop] === 'object') {
        // if the value is an object, recursively call the function on it
        this.removeUndefinedValues(obj[prop]);
        // if the object has no more properties, delete the key
        if (Object.keys(obj[prop]).length === 0) {
          delete obj[prop];
        }
      } else if (obj[prop] === undefined) {
        // remove the key-value pair if the value is undefined
        delete obj[prop];
      }
    }
  }

  async findFilteredPosts(
    searchQuery?: SearchQueryDto,
    filterDto?: PostFilterDto,
    take?: number,
    skip?: number,
  ) {
    const where = {
      AND: [
        {
          OR: [
            { title: { contains: searchQuery?.search } },
            { description: { contains: searchQuery?.search } },
            { user: { firstName: { contains: searchQuery?.search } } },
            { user: { lastName: { contains: searchQuery?.search } } },
          ],
        },
        { address: { contains: searchQuery?.location } },
        { type: filterDto?.type },
        { priority: filterDto?.priority },
        { user: { postcode: filterDto?.postcode } },
        { jobStartDate: filterDto?.jobStartDate },
        { price: { contains: filterDto?.price } },
        { contract: filterDto?.contract },
        { paiement: filterDto?.paiement },
        {
          extra: (filterDto?.extra && filterDto.extra.join(',')) || undefined,
        },
        {
          expirationDate: {
            gt: new Date().toISOString(),
          },
        },
      ],
    };
    this.removeUndefinedValues(where);
    where.AND = where.AND
      ? where.AND.filter((criteria) => Object.keys(criteria).length > 0)
      : where.AND;
    console.log(where);
    const [posts, totalCount] = await Promise.all([
      this.prisma.post.findMany({
        where,
        take,
        skip,
        include: {
          user: true,
        },
      }),
      this.prisma.post.count({
        where,
      }),
    ]);
    return {
      totalCount,
      posts,
    };
  }
  /**
   *
   * @param id
   * @returns A post by id
   */
  async findOne(id: string) {
    return await this.prisma.post.findUnique({
      where: {
        id,
      },
      include: {
        user: true,
      },
    });
  }
  /**
   *
   * @param user
   * @param id
   * @param updatePostDto
   * @returns An uppdated post
   * @throws Bad request in case the post already paid
   */
  async update(user: User, id: string, updatePostDto: UpdatePostDto) {
    const currentPost = await this.prisma.post.findUnique({
      where: {
        id,
      },
    });
    if (
      currentPost?.expirationDate &&
      currentPost?.expirationDate.toISOString() > new Date().toISOString()
    )
      throw new BadRequestException('Post already paid');
    return this.prisma.post.update({
      where: {
        id,
      },
      data: {
        ...updatePostDto,
        user: {
          connect: {
            id: user.id,
          },
        },
        extra: updatePostDto.extra ? updatePostDto.extra.join(',') : '',
      },
    });
  }

  async remove(currentuser: User, id: string) {
    const post = await this.prisma.post.findUnique({
      where: {
        id,
      },
      include: {
        user: true,
        bookmarks: true,
      },
    });
    if (post?.user.id != currentuser.id) {
      throw new ForbiddenException('Resource not valid');
    }

    const res = await this.prisma.post.delete({
      where: {
        id,
      },
    });
    return res;
  }

  //SEARCH SERVICE
  async searchPosts(query: string) {
    const sanitizedQuery = query.replace(/[^a-zA-Z0-9\s]/g, '');
    const result = await this.prisma.post.findMany({
      where: {
        OR: [
          { title: { contains: sanitizedQuery } },
          { description: { contains: sanitizedQuery } },
          { user: { firstName: { contains: sanitizedQuery } } },
          { user: { lastName: { contains: sanitizedQuery } } },
        ],
      },
      include: {
        user: true,
      },
    });
    return result;
  }
}
