import { BadRequestException, Injectable } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';
import { User } from '../_gen/prisma-class/user';
import { SendMessageDto } from './dto/send-message.dto';
import { FilerService } from '../digitalocean-filer/digitalocean-filer.service';
import { AllowedFileTypes } from './types/allowedFileTypes'


@Injectable()
export class MessengerService {
  constructor(
    private readonly prisma: PrismaService,
    private readonly filerService: FilerService,
  ) { }
  /**
   *
   * @param user
   * @param sendMessageDto
   * @returns A created or updated Conversation
   * This service create or update a conversation based on the user and the receiver
   */
  async sendMessage(user: User, sendMessageDto: SendMessageDto) {
    const conversationId = await this.prisma.conversation.findFirst({
      where: {
        OR: [
          {
            userAId: user.id,
            userBId: sendMessageDto.receiverId,
          },
          {
            userAId: sendMessageDto.receiverId,
            userBId: user.id,
          },
        ],
      },
    });
    if (conversationId) {
      return await this.prisma.conversation.update({
        where: {
          id: conversationId.id,
        },
        data: {
          messages: {
            create: {
              text: sendMessageDto.content,
              authorId: user.id,
            },
          },
        },
        include: {
          messages: {
            orderBy: {
              createdAt: 'asc',
            },
            include: {
              author: true,
            },
          },
        },
      });
    } else {
      return await this.prisma.conversation.create({
        data: {
          userAId: user.id,
          userBId: sendMessageDto.receiverId,
          messages: {
            create: {
              authorId: user.id,
              text: sendMessageDto.content,
            },
          },
        },
        include: {
          messages: {
            orderBy: {
              createdAt: 'asc',
            },
            include: {
              author: true,
            },
          },
        },
      });
    }
  }
  async getConversations(user: User) {
    return await this.prisma.conversation.findMany({
      where: {
        OR: [
          {
            userAId: user.id,
          },
          {
            userBId: user.id,
          },
        ],
      },
      include: {
        messages: {
          orderBy: {
            createdAt: 'asc',
          },
          include: {
            author: true,
          },
        },
        userA: {
          include: {
            profile: true,
          },
        },
        userB: {
          include: {
            profile: true,
          },
        },
      },
      take: 10,
      orderBy: {
        updatedAt: 'desc',
      },
    });
  }

  async readMessages(user: User, conversationId: string) {
    await this.prisma.conversation.update({
      where: {
        id: conversationId,
      },
      data: {
        messages: {
          updateMany: {
            where: {
              authorId: {
                not: user.id,
              },
              read: false,
            },
            data: {
              read: true,
            },
          },
        },
      },
    });
    return "done"
  }
  async sendFile(
    files: Array<Express.Multer.File>,
    user: User,
    receiverId: string,
  ) {
    const file = files[0];
    if (!file) {
      throw new BadRequestException('No file provided');
    }
    if (!AllowedFileTypes.includes(file.mimetype)) {
      throw new BadRequestException(`Check your file ${file.originalname}`);
    }
    if (file.size > 4242880) {
      throw new BadRequestException('File too large');
    }
    const fileLink = await this.filerService.uploadFile(file, true);
    const Uploadedfile = await this.prisma.resume.create({
      data: {
        resumeUrl: fileLink,
        ownerId: user.id,
      },
    });

    const conversationId = await this.prisma.conversation.findFirst({
      where: {
        OR: [
          {
            userAId: user.id,
            userBId: receiverId,
          },
          {
            userAId: receiverId,
            userBId: user.id,
          },
        ],
      },
    });
    if (conversationId) {
      return await this.prisma.conversation.update({
        where: {
          id: conversationId.id,
        },
        data: {
          messages: {
            create: {
              text: Uploadedfile.resumeUrl,
              authorId: user.id,
              file: true
            },
          },
        },
        include: {
          messages: {
            orderBy: {
              createdAt: 'asc',
            },
            include: {
              author: true,
            },
          },
        },
      });
    } else {
      return await this.prisma.conversation.create({
        data: {
          userAId: user.id,
          userBId: receiverId,
          messages: {
            create: {
              authorId: user.id,
              text: Uploadedfile.resumeUrl,
              file: true
            },
          },
        },
        include: {
          messages: {
            orderBy: {
              createdAt: 'asc',
            },
            include: {
              author: true,
            },
          },
        },
      });
    }
  }
}
