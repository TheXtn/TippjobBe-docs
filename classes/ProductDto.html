import { ApiProperty } from '@nestjs/swagger';
class ProductDto {
  @ApiProperty({
    description: 'The ID of the product',
    example: 'prod_NuDO1SxnKT98Li',
  })
  id: string;

  @ApiProperty({
    description: 'The object type of the product',
    example: 'product',
  })
  object: string;

  @ApiProperty({
    description: 'Indicates if the product is active',
    example: true,
  })
  active: boolean;

  @ApiProperty({
    type: [String],
    description: 'List of attributes',
    example: [],
  })
  attributes: string[];

  @ApiProperty({
    description: 'The timestamp when the product was created',
    example: 1684247686,
  })
  created: number;

  @ApiProperty({
    description: 'The default price ID of the product',
    example: 'price_1N8OxeGfeNo2kU7nAZl8S0gI',
  })
  default_price: string;

  @ApiProperty({ description: 'The description of the product', example: null })
  description: string | null;

  @ApiProperty({ type: [String], description: 'List of images', example: [] })
  images: string[];

  @ApiProperty({
    description: 'Indicates if the product is in live mode',
    example: false,
  })
  livemode: boolean;

  @ApiProperty({
    description: 'Metadata associated with the product',
    example: { period: '72' },
  })
  metadata: { [key: string]: any };

  @ApiProperty({
    description: 'The name of the product',
    example: 'OFFER_SERVICE/URGENT',
  })
  name: string;

  @ApiProperty({
    description: 'The package dimensions of the product',
    example: null,
  })
  package_dimensions: any | null;

  @ApiProperty({
    description: 'Indicates if the product is shippable',
    example: null,
  })
  shippable: boolean | null;

  @ApiProperty({
    description: 'The statement descriptor of the product',
    example: null,
  })
  statement_descriptor: string | null;

  @ApiProperty({ description: 'The tax code of the product', example: null })
  tax_code: string | null;

  @ApiProperty({ description: 'The type of the product', example: 'service' })
  type: string;

  @ApiProperty({ description: 'The unit label of the product', example: null })
  unit_label: string | null;

  @ApiProperty({
    description: 'The timestamp when the product was last updated',
    example: 1684248753,
  })
  updated: number;

  @ApiProperty({ description: 'The URL of the product', example: null })
  url: string | null;
}

export class ProductListResponseDto {
  @ApiProperty({
    description: 'The object type of the response',
    example: 'list',
  })
  object: string;

  @ApiProperty({ type: [ProductDto], description: 'List of products' })
  data: ProductDto[];

  @ApiProperty({
    description: 'Indicates if there are more products available',
    example: true,
  })
  has_more: boolean;

  @ApiProperty({
    description: 'The URL of the response',
    example: '/v1/products',
  })
  url: string;
}
