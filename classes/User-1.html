import { EmailVerification } from './email_verification';
import { RefreshToken } from './refresh_token';
import { ResetPassword } from './reset_password';
import { Profile } from './profile';
import { EmailChange } from './email_change';
import { Post } from './post';
import { Bookmarks } from './bookmarks';
import { Conversation } from './conversation';
import { Message } from './message';
import { ProfileBookmarks } from './profile_bookmarks';
import { Gender, Type, Visibility } from '@prisma/client';
import { ApiProperty, ApiPropertyOptional } from '@nestjs/swagger';

export class User {
  @ApiProperty({ type: String })
  id: string;

  @ApiProperty({ type: String })
  email: string;

  @ApiProperty({ type: String })
  firstName: string;

  @ApiProperty({ type: String })
  lastName: string;

  @ApiProperty({ enum: Gender, enumName: 'Gender' })
  gender: Gender;

  @ApiProperty({ enum: Type, enumName: 'Type' })
  userType: Type = Type.INDIVIDUAL;

  @ApiProperty({ type: Date })
  birthDate: Date;

  @ApiPropertyOptional({ type: String })
  address?: string;

  @ApiPropertyOptional({ type: String })
  companyName?: string;

  @ApiProperty({ type: String })
  postcode: string;

  @ApiProperty({ type: String })
  password: string;

  @ApiProperty({ type: Date })
  createdAt: Date;

  @ApiProperty({ type: Date })
  updatedAt: Date;

  @ApiPropertyOptional({ type: () => EmailVerification })
  emailVerification?: EmailVerification;

  @ApiProperty({ type: Boolean })
  emailVerified: boolean;

  @ApiPropertyOptional({ type: () => RefreshToken })
  refreshToken?: RefreshToken;

  @ApiPropertyOptional({ type: () => ResetPassword })
  resetPassword?: ResetPassword;

  @ApiPropertyOptional({ type: () => Profile })
  profile?: Profile;

  @ApiProperty({ enum: Visibility, enumName: 'Visibility' })
  visibility: Visibility = Visibility.PUBLIC;

  @ApiProperty({ enum: Visibility, enumName: 'Visibility' })
  uploadsVisibility: Visibility = Visibility.PUBLIC;

  @ApiProperty({ type: Boolean })
  newsletter: boolean;

  @ApiPropertyOptional({ type: () => EmailChange })
  EmailChange?: EmailChange;

  @ApiProperty({ isArray: true, type: () => Post })
  posts: Post[];

  @ApiProperty({ isArray: true, type: () => Bookmarks })
  bookmarks: Bookmarks[];

  @ApiProperty({ isArray: true, type: () => Conversation })
  conversationsA: Conversation[];

  @ApiProperty({ isArray: true, type: () => Conversation })
  conversationsB: Conversation[];

  @ApiProperty({ isArray: true, type: () => Message })
  messages: Message[];

  @ApiProperty({ isArray: true, type: () => ProfileBookmarks })
  bookmarkingUsers: ProfileBookmarks[];

  @ApiProperty({ isArray: true, type: () => ProfileBookmarks })
  bookmarkedUsers: ProfileBookmarks[];
}
