import { ApiProperty, ApiPropertyOptional } from '@nestjs/swagger';
import {
  JobContract,
  JobExtra,
  JobPaiement,
  JobPriority,
  JobType,
} from '@prisma/client';
import { IsEnum, IsOptional, IsString } from 'class-validator';
export class CreatePostDto {
  @ApiProperty({ enum: JobType, enumName: 'JobType' })
  @IsEnum(JobType)
  type: JobType;

  @ApiProperty({ enum: JobPriority, enumName: 'JobPriority' })
  @IsEnum(JobPriority)
  priority: JobPriority;

  @ApiProperty({ enum: JobContract, enumName: 'JobContract' })
  @IsEnum(JobContract)
  contract: JobContract;

  @ApiProperty({ type: String })
  @IsString()
  title: string;

  @ApiPropertyOptional({ type: String })
  @IsString()
  description: string;

  @ApiPropertyOptional({ isArray: true, enum: JobExtra, enumName: 'JobExtra' })
  @IsEnum(JobExtra, { each: true })
  @IsOptional()
  extra: string[];

  @ApiProperty({ type: Date })
  @IsString()
  jobStartDate: Date;

  @ApiProperty({ enum: JobPaiement, enumName: 'JobPaiement' })
  @IsEnum(JobPaiement)
  @IsOptional()
  paiement?: JobPaiement;

  @ApiPropertyOptional({ type: String })
  @IsString()
  @IsOptional()
  address?: string;

  @ApiPropertyOptional({ type: String })
  @IsString()
  @IsOptional()
  contact?: string;

  @ApiPropertyOptional({ type: String })
  @IsOptional()
  @IsString()
  image?: string;

  @ApiPropertyOptional({
    type: String,
  })
  @IsOptional()
  @IsString()
  price?: string;
}
