import { User } from './user';
import { Bookmarks } from './bookmarks';
import { JobType, JobPriority, JobContract, JobPaiement } from '@prisma/client';
import { ApiProperty, ApiPropertyOptional } from '@nestjs/swagger';

export class Post {
  @ApiProperty({ type: String })
  id: string;

  @ApiProperty({ enum: JobType, enumName: 'JobType' })
  type: JobType;

  @ApiProperty({ enum: JobPriority, enumName: 'JobPriority' })
  priority: JobPriority;

  @ApiProperty({ enum: JobContract, enumName: 'JobContract' })
  contract: JobContract;

  @ApiProperty({ type: String })
  title: string;

  @ApiProperty({ type: String })
  description: string;

  @ApiPropertyOptional({ type: String })
  extra?: string;

  @ApiProperty({ type: Date })
  jobStartDate: Date;

  @ApiPropertyOptional({ enum: JobPaiement, enumName: 'JobPaiement' })
  paiement?: JobPaiement;

  @ApiPropertyOptional({ type: String })
  address?: string;

  @ApiPropertyOptional({ type: String })
  contact?: string;

  @ApiPropertyOptional({ type: String })
  image?: string;

  @ApiPropertyOptional({ type: String })
  price?: string;

  @ApiPropertyOptional({ type: Date })
  expirationDate?: Date;

  @ApiProperty({ type: String })
  userId: string;

  @ApiProperty({ type: () => User })
  user: User;

  @ApiProperty({ isArray: true, type: () => Bookmarks })
  bookmarks: Bookmarks[];
}
